#!/bin/bash

# Exit on any error
set -e

# Directory for log files
LOG_DIR="/var/log/kali-update"
LOG_FILE="$LOG_DIR/kali-update-$(date '+%Y%m%d-%H%M%S').log"

# Create log directory if it doesn't exist
mkdir -p "$LOG_DIR"

# Redirect output to both console and log file
exec > >(tee -a "$LOG_FILE") 2>&1

# Function to log messages
log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1"
}

# Clean up old logs, keeping only the last 3
log "Cleaning up old log files..."
find "$LOG_DIR" -name "kali-update-*.log" -type f | sort -r | tail -n +4 | xargs -I {} rm -f {}

# Check if script is run as root
if [ "$EUID" -ne 0 ]; then
    log "ERROR: This script must be run as root. Use sudo."
    exit 1
fi

# Check for internet connectivity
log "Checking internet connectivity..."
if ! ping -c 1 8.8.8.8 > /dev/null 2>&1; then
    log "ERROR: No internet connection. Please check your network."
    exit 1
fi

# Check available disk space (at least 1GB required)
log "Checking available disk space..."
if [ $(df / --output=avail | tail -1) -lt 1048576 ]; then
    log "ERROR: Insufficient disk space (<1GB). Free up space and try again."
    exit 1
fi

# Configure dpkg
log "Configuring dpkg..."
if ! dpkg --configure -a; then
    log "ERROR: Failed to configure dpkg."
    exit 1
fi

# Install missing dependencies
log "Installing missing dependencies..."
if ! apt install -f -y; then
    log "ERROR: Failed to install missing dependencies."
    exit 1
fi

# Update and upgrade packages
log "Updating package lists..."
if ! apt update -y; then
    log "ERROR: Failed to update package lists."
    exit 1
fi

log "Upgrading packages..."
if ! apt upgrade -y; then
    log "ERROR: Failed to upgrade packages."
    exit 1
fi

log "Performing distribution upgrade..."
if ! apt dist-upgrade -y; then
    log "ERROR: Failed to perform distribution upgrade."
    exit 1
fi

# Remove unnecessary packages
log "Removing unnecessary packages..."
if ! apt --purge autoremove -y; then
    log "WARNING: Failed to remove unnecessary packages. Continuing..."
fi

# Clean package cache
log "Cleaning package cache..."
if ! apt autoclean -y; then
    log "WARNING: Failed to clean package cache. Continuing..."
fi

# Check for reboot requirement
if [ -f /var/run/reboot-required ]; then
    log "WARNING: A system reboot is required to complete updates."
fi

log "Update completed successfully!"
exit 0
